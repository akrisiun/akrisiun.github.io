using System;
using Mono.Math;

namespace Sharpen
{
	internal class DSAParams
	{
		BigInteger g, p, q;
		
		public DSAParams (BigInteger g, BigInteger p, BigInteger q)
		{
			this.g = g;
			this.p = p;
			this.q = q;
		}
		
		public BigInteger GetG ()
		{
			return g;
		}
		
		public BigInteger GetP ()
		{
			return p;
		}
		
		public BigInteger GetQ ()
		{
			return q;
		}
	}
}
