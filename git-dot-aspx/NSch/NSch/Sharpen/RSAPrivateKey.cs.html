using System;
using Mono.Math;
using System.Security.Cryptography;

namespace Sharpen
{
	internal class RSAPrivateKey: PrivateKey
	{
		public RSAParameters Parameters { get; private set; }
		
		public RSAPrivateKey (RSAParameters pars)
		{
			Parameters = pars;
		}
		
		public BigInteger GetPrivateExponent ()
		{
			return new BigInteger (Parameters.D);
		}
		
		public BigInteger GetModulus ()
		{
			return new BigInteger (Parameters.Modulus);
		}
	}
}
