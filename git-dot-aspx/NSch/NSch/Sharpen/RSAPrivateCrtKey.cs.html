using System;
using Mono.Math;
using System.Security.Cryptography;

namespace Sharpen
{
	internal class RSAPrivateCrtKey: RSAPrivateKey
	{
		RSAParameters pars;
		
		public RSAPrivateCrtKey (RSAParameters pars): base (pars)
		{
			this.pars = pars;
		}
		
		public BigInteger GetCrtCoefficient ()
		{
			return new BigInteger (pars.InverseQ);
		}
		
		public BigInteger GetPrimeExponentP ()
		{
			return new BigInteger (pars.DP);
		}
		
		public BigInteger GetPrimeExponentQ ()
		{
			return new BigInteger (pars.DQ);
		}
		
		public BigInteger GetPrimeP ()
		{
			return new BigInteger (pars.P);
		}
		
		public BigInteger GetPrimeQ ()
		{
			return new BigInteger (pars.Q);
		}
	}
}
