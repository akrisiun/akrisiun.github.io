using System;

namespace Sharpen
{
	public abstract class SecretKeyFactory
	{
		public SecretKeyFactory ()
		{
		}
		
		public static SecretKeyFactory GetInstance (string id)
		{
			if (id == "DESede")
				return new DESedeSecretKeyFactory ();
			throw new NotSupportedException ();
		}
		
		public abstract SecretKey GenerateSecret (KeySpec key);
	}
	
	class DESedeSecretKeyFactory: SecretKeyFactory
	{
		public override SecretKey GenerateSecret (KeySpec key)
		{
			DESedeKeySpec ks = (DESedeKeySpec) key;
			return new SecretKey (ks.Key);
		}
	}
}
